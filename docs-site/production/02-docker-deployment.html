Docker Deployment

Containerized deployment for production environments

Dockerfile

dockerfile 
FROM node:20-alpine AS builder
WORKDIR /app
COPY package*.json ./
RUN npm ci --only=production
COPY . .
RUN npm run build

FROM node:20-alpine
WORKDIR /app
COPY --from=builder /app/dist ./dist
COPY --from=builder /app/node_modules ./node_modules
COPY package*.json ./
ENV NODE_ENV=production
EXPOSE 3000
CMD ["node", "dist/daemon.js"]

Building

bash 
# Build image
docker build -t ff-terminal:latest .

# Tag for registry
docker tag ff-terminal:latest registry.example.com/ff-terminal:2.0.0

# Push to registry
docker push registry.example.com/ff-terminal:2.0.0

Docker Compose

yaml 
version: '3.8'
services:
  ff-daemon:
    image: ff-terminal:latest
    container_name: ff-daemon
    restart: unless-stopped
    ports:
      - "3000:3000"
    environment:
      - NODE_ENV=production
      - FF_PORT=3000
      - FF_LOG_LEVEL=info
    volumes:
      - ./data:/app/data
      - ./logs:/app/logs
    networks:
      - ff-network
    healthcheck:
      test: ["CMD", "node", "-e", "require('http').get('http://localhost:3000/health')"]
      interval: 30s
      timeout: 10s
      retries: 3

networks:
  ff-network:
    driver: bridge

Running

bash 
# Start services
docker-compose up -d

# View logs
docker-compose logs -f ff-daemon

# Stop services
docker-compose down

# Restart single service
docker-compose restart ff-daemon
